public int gcd(int p,int q){
      if(q == 0)    return p;
      return gcd(q, p % q);
 }

//最小公倍数(Least Common Multiple)　
public int lcm(int p,int q){
      int pq = p * q;
      return pq / gcd(p,q);//除以公约数

}
